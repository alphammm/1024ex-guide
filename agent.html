agent.py — OpenClaw Agent 核心代码 
from openclaw import Agent, Tool
from exchange_1024 import TradingClient

# 初始化 1024EX 交易客户端
client = TradingClient(
    api_key="your_api_key",
    api_secret="your_api_secret",
    network="arbitrum"
)

# 定义交易工具集
tools = [
    Tool("market_order",  client.market_order,  "市价开仓: 方向/币种/金额/杠杆"),
    Tool("limit_order",   client.limit_order,   "限价挂单: 方向/币种/价格/金额"),
    Tool("close_position", client.close,         "平仓: 币种/比例"),
    Tool("get_positions",  client.positions,     "查询当前所有持仓"),
    Tool("get_price",      client.ticker,        "查询实时价格"),
    Tool("get_balance",    client.balance,       "查询账户余额"),
    Tool("set_tp_sl",      client.set_tp_sl,     "设置止盈止损"),
]

# 创建 AI Agent（支持自然语言交互）
agent = Agent(
    name="1024EX_Trader",
    model="claude-sonnet-4-6",
    tools=tools,
    system_prompt="""你是 1024EX 链上交易助手。
    用户通过 IM 发来交易指令，你负责：
    1. 解析意图（开仓/平仓/查询/分析）
    2. 确认关键参数（方向/金额/杠杆）
    3. 执行交易并返回结果
    高风险操作（大额/高杠杆）需二次确认。"""
)
bot.py — Telegram Bot 接入 
from telegram.ext import Application, MessageHandler, filters

async def handle_message(update, context):
    # 用户的 IM 消息 → Agent 处理 → 返回结果
    user_msg = update.message.text
    result = await agent.run(user_msg)
    await update.message.reply_text(result)

# 启动 Telegram Bot
app = Application.builder().token("TG_BOT_TOKEN").build()
app.add_handler(MessageHandler(filters.TEXT, handle_message))
app.run_polling()
